```python
# Calculate the first 10,000 terms of the series
terms = 10000

total = 0.0

for n in range(terms):
    # Alternating series: +1 for even indices, -1 for odd indices
    if n % 2 == 0:
        total += 1 / (2 * n + 1)
    else:
        total -= 1 / (2 * n + 1)

# Multiply the total by 4
result = total * 4

# Print the result
print(result)
```
